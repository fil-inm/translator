int add(int a, int b) {
    return a + b;
}

int add(int a) {
    return a + 10;
}

// ==============================
// Рекурсия
// ==============================

int fib(int n) {
    if (n == 1 || n == 2) {
        return 1;
    }
    else {
        return fib(n - 1) + fib(n - 2);
    }
}

int fact(int n) {
    if (n <= 1) {
        return 1;
    }
    return n * fact(n - 1);
}

// ==============================
// Массивы
// ==============================


int max(int a, int b) {
    if (a > b) {
        return a;
    }
    return b;
}

    return g(x-1) + 1;
}

int g(int x) {
    if (x < 0) {
        return x;
    }
    return f(x - 1) + 1;
}
